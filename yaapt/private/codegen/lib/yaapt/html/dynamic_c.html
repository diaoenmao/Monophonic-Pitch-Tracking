  374   void eml_li_find(const emxArray_boolean_T *x, emxArray_int32_T *y)
  375   {
  376     int k;
  377     int i;

  381       if (x->data[i - 1]) {
  382         k++;
  383       }
  384     }
  385   
  386     j = y->size[0] * y->size[1];
  387     y->size[0] = 1;
  388     y->size[1] = k;
  389     emxEnsureCapacity((emxArray__common *)y, j, (int)sizeof(int));

  391     for (i = 1; i <= x->size[1]; i++) {
  392       if (x->data[i - 1]) {
