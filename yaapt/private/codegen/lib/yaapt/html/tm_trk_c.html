  493       for (i27 = 0; i27 < loop_ub; i27++) {
  494         r27->data[i27] = i27;
  495       }
  496   
  497       unnamed_idx_1 = r27->size[0];
  498       for (i27 = 0; i27 < unnamed_idx_1; i27++) {
  499         Pitch->data[r27->data[i27] + Pitch->size[0] * n] = b_y->data[i27];
  500       }
  501   
  502       loop_ub = Merit->size[0];
  503       i27 = r27->size[0];
  504       r27->size[0] = loop_ub;
  505       emxEnsureCapacity((emxArray__common *)r27, i27, (int)sizeof(int));
  506       for (i27 = 0; i27 < loop_ub; i27++) {
  507         r27->data[i27] = i27;

  524     /*    The following lines increase merit for peaks which are very */
  525     /*    close to in frequency to those peaks which are close to */
  526     /*    smoothed F0 track from spectrogram,   and decrease merit for peaks */
  527     /*    which are not close to the smoothed F0 track obtained from */
  528     /*    spectrogram */
  529     i = 0;
  530     emxInit_real_T(&diff, 2);
  531     emxInit_real_T(&r28, 2);
  532     while (i <= (int)Prm_nccf_maxcands - 1) {
  533       loop_ub = Pitch->size[1];
  534       i27 = temp_max->size[0] * temp_max->size[1];
  535       temp_max->size[0] = 1;
  536       temp_max->size[1] = loop_ub;
  537       emxEnsureCapacity((emxArray__common *)temp_max, i27, (int)sizeof(double));
  538       for (i27 = 0; i27 < loop_ub; i27++) {
  539         temp_max->data[temp_max->size[0] * i27] = Pitch->data[i + Pitch->size[0] *
  540           i27] - SPitch->data[SPitch->size[0] * i27];
  541       }
  542   
  543       i27 = diff->size[0] * diff->size[1];
  544       diff->size[0] = 1;
  545       diff->size[1] = temp_max->size[1];
  546       emxEnsureCapacity((emxArray__common *)diff, i27, (int)sizeof(double));
  547       for (k = 0; k + 1 <= temp_max->size[1]; k++) {
  548         diff->data[k] = fabs(temp_max->data[k]);
  549       }
  550   
  551       i27 = temp_max->size[0] * temp_max->size[1];
  552       temp_max->size[0] = 1;
  553       temp_max->size[1] = diff->size[1];
  554       emxEnsureCapacity((emxArray__common *)temp_max, i27, (int)sizeof(double));
  555       loop_ub = diff->size[1];
  556       for (i27 = 0; i27 < loop_ub; i27++) {
  557         temp_max->data[i27] = 0.0;
  558       }
  559   
  560       for (j = 0; j < diff->size[1]; j++) {
  561         X = diff->data[j];
  562         loop_ub = y->size[0];
  563         for (i27 = 0; i27 < loop_ub; i27++) {
  564           tmp_data[i27] = (signed char)(X < y->data[i27]);
  565         }
  566   
  567         temp_max->data[j] = tmp_data[0];
  568       }
  569   
  570       /*      match1 =  (diff < freq_thresh); */
  571       i27 = temp_min->size[0] * temp_min->size[1];
  572       temp_min->size[0] = 1;
  573       temp_min->size[1] = diff->size[1];
  574       emxEnsureCapacity((emxArray__common *)temp_min, i27, (int)sizeof(double));
  575       loop_ub = diff->size[1];
  576       for (i27 = 0; i27 < loop_ub; i27++) {
  577         temp_min->data[i27] = 0.0;
  578       }
  579   
  580       for (j = 0; j < diff->size[1]; j++) {
  581         if (y->size[0] == 0) {
  582           unnamed_idx_1 = 0;
  583         } else {
  584           X = 1.0 / freq_thresh_data[0] * diff->data[j];
  585           unnamed_idx_1 = 1;
  586           y_data[0] = X;
  587         }
  588   
  589         i27 = b_y->size[0] * b_y->size[1];
  590         b_y->size[0] = 1;
  591         b_y->size[1] = unnamed_idx_1;

  593         X = temp_max->data[j];
  594         for (i27 = 0; i27 < unnamed_idx_1; i27++) {

  596         }
  597   
  598         temp_min->data[j] = b_y->data[0];
  599       }

  601       /*      match  =  ((1 - (diff./freq_thresh)) .* match1); */
  602       unnamed_idx_1 = Merit->size[1];
  603       i27 = r28->size[0] * r28->size[1];
  604       r28->size[0] = 1;
  605       r28->size[1] = unnamed_idx_1;
  606       emxEnsureCapacity((emxArray__common *)r28, i27, (int)sizeof(double));
  607       for (i27 = 0; i27 < unnamed_idx_1; i27++) {
  608         r28->data[r28->size[0] * i27] = (1.0 + Prm_merit_boost) * Merit->data[i +
  609           Merit->size[0] * i27] * temp_min->data[temp_min->size[0] * i27];
  610       }
  611

  615       }

  617       i++;
  618     }
